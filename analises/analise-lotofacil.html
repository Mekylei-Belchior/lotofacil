def grafico_linha_plt(x, y, title, xlabel, ylabel):
    """
    Funo para criar um grfico de linhas utilizando o Matplotlib.
    
    Parmetros:
    x      -> Valores do eixo (x)
    y      -> Valores do eixo (y)
    title  -> Ttulo do grfico
    xlabel -> Rtulo do eixo (x)
    ylabel -> Rtulo do eixo (y)
    
    """
    
    # Define as dimensses do grfico
    plt.figure(figsize=(15, 5))
    
    # Define o ttulo, rtulo x, rtulo y
    plt.title(title)
    plt.xlabel(xlabel)
    plt.ylabel(ylabel)
    
    # Cria o grfico
    plt.plot(x, y, color='purple', marker='o', linestyle='dashed', linewidth=1.5, markersize=4)
    # Formata a linha de grade do grfico
    plt.grid(linestyle='--', which='major', color='grey', alpha=.25)
    # Visualiza o grfico
    plt.show()
    
    
def grafico_linha_go(x, y, title, xlabel, ylabel):
    """
    Funo para criar um grfico de linhas utilizando o Plotly.
    
    Parmetros:
    x      -> Valores do eixo (x)
    y      -> Valores do eixo (y)
    title  -> Ttulo do grfico
    xlabel -> Rtulo do eixo (x)
    ylabel -> Rtulo do eixo (y)
    
    """
    
    # Criando o grfico
    trace = go.Scatter(x=x, y=y, mode='markers+lines')

    # Armazenando grfico em uma lista
    grafico = [trace]

    # Criando Layout
    layout = go.Layout(title=title, xaxis={'title': xlabel}, yaxis={'title': ylabel})

    # Criando figura que ser exibida
    fig = go.Figure(data=grafico, layout=layout)

    # Visualiza o grfico
    py.iplot(fig)
